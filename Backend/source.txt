func dividir(_ numero1: Int, por numero2: Int) -> Int {
    // Usamos guard para verificar que el divisor no sea cero
    guard numero2 != 0 else {
        // Si el divisor es cero, retornamos nil y manejamos el error
        return -1
    }
    
    // Realizamos la división si el divisor no es cero
    let resultado = numero1 / numero2
    return resultado
}

// Ejemplo de uso
let resultado = dividir(10, por: 0)
if resultado !=-1 {
    print("El resultado de la división es:", resultado)
} else {
    print("No se puede dividir por cero.")
}
